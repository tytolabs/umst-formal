\section*{1\quad Gate as Natural Transformation}
\headline{Interpretation}
The gate $G:\TS \times \TS \to \Bool$ is material-agnostic: changing material labels
via $f:\MC \to \MC'$ commutes with state construction before gate evaluation.

\[
\begin{tikzcd}[column sep=3.2em, row sep=2.5em]
\MC \times \MC
  \arrow[r, "F_m \times F_m"]
  \arrow[d, "f \times f"']
& \TS \times \TS
  \arrow[r, "G"]
  \arrow[d, "F(f) \times F(f)"]
& \Bool
  \arrow[d, equal]
\\
\MC' \times \MC'
  \arrow[r, "F_{m'} \times F_{m'}"']
& \TS \times \TS
  \arrow[r, "G"']
& \Bool
\end{tikzcd}
\]

\section*{2\quad DIB as Kleisli Composition}
\headline{Interpretation}
Discovery, Invention, and Build are composed in $\Kl(\StateT\;\TS\;\mathrm{IO})$.
The structure provides sequential composition guarantees for gate-checked transitions.

\[
\begin{tikzcd}[column sep=2.5em, row sep=2.3em]
& \TS \arrow[dr, "I"]
\\
\TS \arrow[ur, "D"] \arrow[rr, "B \circ I \circ D"', bend right=15]
& & \TS \arrow[ll, "B", bend right=15]
\end{tikzcd}
\qquad
\begin{aligned}
D,I,B &: \TS \to \Kl(\TS), \\
\DIB &:= B \circ_{\Kl} I \circ_{\Kl} D.
\end{aligned}
\]

\section*{3\quad Mass as Monoidal Coherence}
\headline{Interpretation}
For independent subsystem composition $\oplus$, density behaves as an additive map up
to tolerance $\delta$, used as the gate's conservation band.

\[
\begin{tikzcd}[column sep=3.2em, row sep=2.3em]
\TS \otimes \TS
  \arrow[r, "\rho \otimes \rho"]
  \arrow[d, "\oplus"']
& \mathbb{Q} \times \mathbb{Q}
  \arrow[d, "{+}"]
\\
\TS
  \arrow[r, "\rho"']
& \mathbb{Q}
\end{tikzcd}
\qquad
\left|\rho(s_1 \oplus s_2)-(\rho(s_1)+\rho(s_2))\right| < \delta
\]
